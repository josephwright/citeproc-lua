\documentclass{article}

\usepackage{citation-style-language}

\cslsetup{style = apa}
\addbibresource{example.json}


\begin{document}

% Single cite item in a citation
\cite{ITEM-1}

% Multiple cite items in a citation
\cite{ITEM-1, ITEM-2}

% Citation with page locator
\cite[page = 6]{ITEM-1}

% Citation with section locator
\cite[section = 2.3]{ITEM-1}

% Citation with implicit page locator
\cite[6]{ITEM-1}

% Citation with prefix and suffix
\cite[prefix = {prefix }, suffix = { suffix}]{ITEM-1}

% Citation with implicit prefix and suffix
\cite[prefix][suffix]{ITEM-1}

% Citation with implicit suffix
\cite[suffix]{ITEM-1}

% Citation with implicit prefix
\cite[prefix][]{ITEM-1}

% Multiple cite items with options
\cites[See][page = 6]{ITEM-1}[figure = 2]{ITEM-2}

% Uncited item
\nocite{ITEM-3}

% Uncited all items
\nocite{*}

% Print the reference list
\printbibliography

\end{document}
