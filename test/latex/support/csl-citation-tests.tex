\ExplSyntaxOn
  \NewDocumentCommand \showcitation { }
    { \tl_show:N \l__csl_citation_tl }
\ExplSyntaxOff


\TEST{Single cite item in a citation}{
  \cite{ITEM-1}
  \showcitation
}

\TEST{Multiple cite items in a citation}{
  \cite{ITEM-1, ITEM-2}
  \showcitation
}

\TEST{Citation with page locator}{
  \cite[page=6]{ITEM-1}
  \showcitation
}

\TEST{Citation with section locator}{
  \cite[section=2.3]{ITEM-1}
  \showcitation
}

\TEST{Citation with implicit page locator}{
  \cite[6]{ITEM-1}
  \showcitation
}

\TEST{Citation with prefix and suffix}{
  \cite[prefix][suffix]{ITEM-1}
  \showcitation
}

\TEST{Citation with suffix}{
  \cite[suffix]{ITEM-1}
  \showcitation
}

\TEST{Citation with prefix}{
  \cite[prefix][]{ITEM-1}
  \showcitation
}
